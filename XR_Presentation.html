def get_rotation_matrix(p, y, r):
    cx, sx = math.cos(p), math.sin(p)
    cy, sy = math.cos(y), math.sin(y)
    cz, sz = math.cos(r), math.sin(r)
    return np.array([
        [cy*cz + sy*sx*sz, -cy*sz + sy*sx*cz, sy*cx],
        [cx*sz, cx*cz, -sx],
        [-sy*cz + cy*sx*sz, sy*sz + cy*sx*cz, cy*cx]
    ])

def project_point(pt, f, cx, cy):
    # x = (X * f / Z) + cx
    # y = (Y * f / Z) + cy
    return (
        int((pt[0] * f / pt[2]) + cx),
        int((pt[1] * f / pt[2]) + cy)
    ) if pt is not None and pt[2] > 10.0 else None
[SAFETY] 設定 `pt[2] > 10.0` (Z軸大於 10mm) 是為了避免物體過於靠近相機時造成 除以零 (Divide by Zero) 的數學崩潰。

def intersect(p1, p2, p3, p4):
    # 計算 2D 外積
    c = lambda o, a, b: \
        (a[0] - o[0]) * (b[1] - o[1]) - \
        (a[1] - o[1]) * (b[0] - o[0])

    return (
        (c(p3, p4, p1) * c(p3, p4, p2) < 0) and
        (c(p1, p2, p3) * c(p1, p2, p4) < 0)
    )

def __call__(self, x, dt):
    self.dx_prev = a_d * ((x - self.x_prev) / dt) + (1.0 - a_d) * self.dx_prev

    cutoff = self.mincutoff + self.beta * math.hypot(*self.dx_prev)

    a = 1.0 / (1.0 + 1.0 / (2 * math.pi * cutoff * dt))
    self.x_prev = a * x + (1.0 - a) * self.x_prev
    return self.x_prev
