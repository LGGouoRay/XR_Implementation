1€ Filter (One Euro)

實時降低高頻抖動，保留高速響應，大幅提升面板拖曳平滑度。

class OneEuroFilter: def __init__(self, mincutoff=1.0, beta=0.007): # 動態調整截止頻率以對應速度 self.mincutoff, self.beta = mincutoff, beta def __call__(self, x, dt): # 計算動態截止頻率 Cutoff cutoff = self.mincutoff + self.beta * math.hypot(*self.dx) a = 1.0 / (1.0 + 1.0 / (2 * math.pi * cutoff * dt)) return a * x + (1.0 - a) * self.prev_x
